import java.util.HashMap;

    class Person{
        private String navn;
        private HashMap eideFilmer = new HashMap();
        private HashMap utlanteFilmer = new HashMap();

        public Person(String navn){
            this.navn = navn;
        }

        public String toString(){
            return navn;
        }

        public int antallFilmer(){
            return eideFilmer.size();
        }

        // Legg merke til returverdien, hvor man returnerer HashMap'et i sin helhet
        public HashMap hentEide(){
            return eideFilmer;
        }

        public boolean eierFilm(String navn){ // Metode for å sjekke feil Case 1 i laan()-metoden
            for (DVD film : eideFilmer.values()){
                if (film.toString().equals(navn)){
                    return true;
                }
            }
            return false;
        }

        public void leggTilFilm(DVD nyFilm){
            eideFilmer.put(nyFilm.toString(), nyFilm);
        }


        // Under ligger metoder som har med utlånte filmer å gjøre


        public int antallUtlant(){
            return utlanteFilmer.size();
        }

        public void leggTilLant(DVD nyligLant){
            utlanteFilmer.put(nyligLant.toString(), nyligLant);
        }

        public void slettLantFilm(String navn){
            utlanteFilmer.remove(navn);
        }

        public boolean erUtlant(String navn){
            if (utlanteFilmer.containsKey(navn)){
                return utlanteFilmer.get(navn).erUtlant();
            }
            return false;
        }
    }
